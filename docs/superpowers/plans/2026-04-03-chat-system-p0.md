# Chat System P0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 1:1 text chat system with friend requests, real-time messaging, and read markers for the LINE clone.

**Architecture:** Zero custom mutators handle all cross-table operations (`acceptFriendship`, `sendMessage`). Four new DB tables (`friendship`, `chat`, `chatMember`, `message`) sync via Zero WAL replication. Navigation restructures to a combined "Chats + Friends" tab with pill toggle, replacing the feed starter template.

**Tech Stack:** Zero (`@rocicorp/zero` 0.26.2) + `on-zero` for models/queries/mutations, Drizzle for DB schema + migrations, Tamagui for UI, OneJS for routing.

**Spec:** `docs/superpowers/specs/2026-04-03-chat-system-p0-design.md`

```typescript
import { number, string, table } from '@rocicorp/zero'
import { mutations, serverWhere } from 'on-zero'

import type { TableInsertRow } from 'on-zero'

export type ChatMember = TableInsertRow<typeof schema>

export const schema = table('chatMember')
  .columns({
    id: string(),
    chatId: string(),
    userId: string(),
    lastReadMessageId: string().optional(),
    lastReadAt: number().optional(),
    joinedAt: number(),
  })
  .primaryKey('id')

// A user can read chatMember records if:
// - it's their own record, OR
// - the record belongs to a chat they are also a member of (needed for read markers)
const chatMemberPermission = serverWhere('chatMember', (eb, auth) => {
  return eb.or(
    eb.cmp('userId', auth?.id || ''),
    eb.exists('chat', (q) =>
      q.exists('members', (mq) => mq.where('userId', auth?.id || '')),
    ),
  )
})

export const mutate = mutations(schema, chatMemberPermission, {
  // Custom update for read marker — only allows updating lastReadMessageId/lastReadAt
  markRead: async ({ authData, tx }, data: { id: string; lastReadMessageId: string; lastReadAt: number }) => {
    if (!authData) throw new Error('Unauthorized')
    await tx.mutate.chatMember.update({
      id: data.id,
      lastReadMessageId: data.lastReadMessageId,
      lastReadAt: data.lastReadAt,
    })
  },
})
```

- [ ] **Update the `mutate` export in `packages/zero-schema/src/models/friendship.ts` to add the `accept` custom mutator:**

```typescript
import { number, string, table } from '@rocicorp/zero'
import { mutations, serverWhere } from 'on-zero'

import type { TableInsertRow } from 'on-zero'

export type Friendship = TableInsertRow<typeof schema>

export const schema = table('friendship')
  .columns({
    id: string(),
    requesterId: string(),
    addresseeId: string(),
    status: string(),
    createdAt: number(),
    updatedAt: number(),
  })
  .primaryKey('id')

export const friendshipPermission = serverWhere('friendship', (eb, auth) => {
  return eb.or(
    eb.cmp('requesterId', auth?.id || ''),
    eb.cmp('addresseeId', auth?.id || ''),
  )
})

export const mutate = mutations(schema, friendshipPermission, {
  accept: async (
    { authData, tx },
    args: {
      friendshipId: string
      chatId: string
      member1Id: string  // caller (addressee) chatMember id
      member2Id: string  // requester chatMember id
      requesterId: string
    },
  ) => {
    if (!authData) throw new Error('Unauthorized')

    const now = Date.now()

    // Update friendship status
    await tx.mutate.friendship.update({
      id: args.friendshipId,
      status: 'accepted',
      updatedAt: now,
    })

    // Create the chat room (omit optional fields — Zero maps undefined → NULL)
    await tx.mutate.chat.insert({
      id: args.chatId,
      type: 'direct',
      createdAt: now,
    })

    // Create chatMember for the addressee (current user)
    await tx.mutate.chatMember.insert({
      id: args.member1Id,
      chatId: args.chatId,
      userId: authData.id,
      joinedAt: now,
    })

    // Create chatMember for the requester
    await tx.mutate.chatMember.insert({
      id: args.member2Id,
      chatId: args.chatId,
      userId: args.requesterId,
      joinedAt: now,
    })
  },
})
```

import { number, string, table } from '@rocicorp/zero'
import { mutations, serverWhere } from 'on-zero'

import type { TableInsertRow } from 'on-zero'

export type Message = TableInsertRow<typeof schema>

export const schema = table('message')
  .columns({
    id: string(),
    chatId: string(),
    senderId: string(),
    type: string(),
    text: string().optional(),
    metadata: string().optional(),
    replyToMessageId: string().optional(),
    createdAt: number(),
  })
  .primaryKey('id')

export const messageReadPermission = serverWhere('message', (eb, auth) => {
  return eb.exists('members', (q) => q.where('userId', auth?.id || ''))
})

export const mutate = mutations(schema, messageReadPermission, {
  send: async (
    { authData, tx },
    message: Message,
  ) => {
    if (!authData) throw new Error('Unauthorized')
    if (message.senderId !== authData.id) throw new Error('Unauthorized')

    // Insert the message
    await tx.mutate.message.insert(message)

    // Update the chat's last message pointer for sorting the chat list
    await tx.mutate.chat.update({
      id: message.chatId,
      lastMessageId: message.id,
      lastMessageAt: message.createdAt,
    })
  },
})

import { chatMembersByChatId } from '@vine/zero-schema/queries/chat'
import { messagesByChatId } from '@vine/zero-schema/queries/message'
import { useAuth } from '~/features/auth/client/authClient'
import { zero, useZeroQuery } from '~/zero/client'

export function useMessages(chatId: string) {
  const { user } = useAuth()
  const userId = user?.id ?? ''

  const [messages, { type }] = useZeroQuery(
    messagesByChatId,
    { chatId },
    { enabled: Boolean(chatId) },
  )

  const [members] = useZeroQuery(
    chatMembersByChatId,
    { chatId },
    { enabled: Boolean(chatId) },
  )

  const isLoading = type === 'unknown'

  // Find the current user's chatMember record (for read marker updates)
  const myMembership = useMemo(
    () => members?.find((m) => m.userId === userId) ?? null,
    [members, userId],
  )

  const otherMember = useMemo(
    () => members?.find((m) => m.userId !== userId) ?? null,
    [members, userId],
  )

  const sendMessage = (text: string) => {
    if (!userId || !text.trim()) return
    // Omit optional fields (metadata, replyToMessageId) — Zero maps undefined → NULL
    zero.mutate.message.send({
      id: crypto.randomUUID(),
      chatId,
      senderId: userId,
      type: 'text',
      text: text.trim(),
      createdAt: Date.now(),
    })
  }

  const markRead = (latestMessageId: string) => {
    if (!myMembership) return
    // Only update if this message is newer than the last read
    zero.mutate.chatMember.markRead({
      id: myMembership.id,
      lastReadMessageId: latestMessageId,
      lastReadAt: Date.now(),
    })
  }

  return {
    messages: messages ?? [],
    isLoading,
    myMembership,
    otherMember,
    sendMessage,
    markRead,
  }
}

function formatChatTime(ts: number): string {
  const date = new Date(ts)
  const now = new Date()
  const diffMs = now.getTime() - date.getTime()
  const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24))

  if (diffDays === 0) {
    return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
  }
  if (diffDays === 1) return '昨天'
  if (diffDays < 7) {
    return ['週日', '週一', '週二', '週三', '週四', '週五', '週六'][date.getDay()]
  }
  return `${date.getMonth() + 1}/${date.getDate()}`
}

```typescript
import { router, useParams } from 'one'
import { memo, useEffect, useRef } from 'react'
import { FlatList } from 'react-native'
import { SafeAreaView } from 'react-native-safe-area-context'
import { SizableText, XStack, YStack } from 'tamagui'

import { useMessages } from '~/features/chat/useMessages'
import { MessageBubble } from '~/features/chat/ui/MessageBubble'
import { MessageInput } from '~/features/chat/ui/MessageInput'
import { useAuth } from '~/features/auth/client/authClient'
import { Avatar } from '~/interface/avatars/Avatar'
import { Button } from '~/interface/buttons/Button'

export const ChatPage = memo(() => {
  const { chatId } = useParams()
  const { user } = useAuth()
  const userId = user?.id ?? ''

  const { messages, isLoading, myMembership, otherMember, sendMessage, markRead } =
    useMessages(chatId as string)

  const flatListRef = useRef<FlatList>(null)

  // Mark as read when entering or when new messages arrive
  useEffect(() => {
    if (messages.length === 0) return
    const lastMessage = messages[messages.length - 1]
    if (!lastMessage) return
    markRead(lastMessage.id)
  }, [messages])

  const handleSend = (text: string) => {
    sendMessage(text)
    // Scroll to bottom after sending
    setTimeout(() => {
      flatListRef.current?.scrollToEnd({ animated: true })
    }, 100)
  }

  const otherUser = otherMember?.user
  const chatName = otherUser?.name ?? '聊天'

  return (
    <SafeAreaView style={{ flex: 1 }} edges={['top']}>
      <YStack flex={1} bg="$background">
        {/* Header */}
        <XStack
          px="$3"
          py="$2"
          items="center"
          gap="$3"
          borderBottomWidth={1}
          borderBottomColor="$color4"
        >
          <Button variant="transparent" onPress={() => router.back()} px="$2">
            ←
          </Button>
          <Avatar size={36} image={otherUser?.image} name={chatName} />
          <SizableText size="$4" fontWeight="600" flex={1}>
            {chatName}
          </SizableText>
        </XStack>

        {/* Messages */}
        <FlatList
          ref={flatListRef}
          data={messages}
          keyExtractor={(item) => item.id}
          renderItem={({ item }) => {
            const isMine = item.senderId === userId
            // "已讀" shown only on my last message if the other member has read it
            const isRead = isMine
              ? (otherMember?.lastReadMessageId != null &&
                  item.createdAt <= (messages.find(m => m.id === otherMember.lastReadMessageId)?.createdAt ?? 0))
              : false

            return (
              <MessageBubble
                text={item.text ?? ''}
                isMine={isMine}
                createdAt={item.createdAt}
                isRead={isRead}
              />
            )
          }}
          onContentSizeChange={() => flatListRef.current?.scrollToEnd({ animated: false })}
          ListEmptyComponent={
            isLoading ? null : (
              <YStack p="$6" items="center">
                <SizableText color="$color9">傳訊息打個招呼吧！</SizableText>
              </YStack>
            )
          }
          style={{ flex: 1 }}
          contentContainerStyle={{ paddingVertical: 8 }}
        />

        {/* Input */}
        <MessageInput onSend={handleSend} />
      </YStack>
    </SafeAreaView>
  )
})
```
